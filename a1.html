The goal of this assignment was to build and code a circuit with a minimum of 3 LEDs blinking independently.

Voltage Calculations

Yellow & Red LEDs:
V = 5, Vdrop = 1.8, I = 20mA
5 - 1.8v = 20mA x Ω
3.2v = 20mA x Ω
3.2v / 20mA = Ω

160 = Ω

Blue LED:
V = 5, Vdrop = 3.3, I = 20mA
5 - 3.3v = 20mA x Ω
1.7v = 20mA x Ω
1.7v / 20mA = Ω

85 = Ω

Code

        
          void setup() {
              pinMode(13, OUTPUT);          //Setting pin 13 to output to an LED
              pinMode(12, OUTPUT);          //Setting pin 12 to output to an LED
              pinMode(11, OUTPUT);          //Setting pin 11 to output to an LED
          }

          int i = 0;                        //Initializing a counter for the program loop

          void loop() {
              while(i < 10) {               //Program loop set to run while i < 10
                delay(500);                 //Adding a .5s delay
                digitalWrite(11, HIGH);     //Turning pin 11 on
                digitalWrite(12, LOW);      //Turning pin 12 off
                delay(500);                 //Waiting for 0.5s before changing the pins
                digitalWrite(12, HIGH);     //Turning pin 12 on
                digitalWrite(13, LOW);      //Turning pin 13 off
                delay(500);                 //Waiting for 0.5s before changing the pins
                digitalWrite(13, HIGH);     //Turning pin 13 on
                digitalWrite(11, LOW);      //Turning pin 11 off
                i++;                        //Incrementing the counter
              }

              delay(500);                   //Initializing a counter for the program loop
              digitalWrite(11, LOW);        //Turning pin 11 off
              digitalWrite(12, LOW);        //Turning pin 12 off
              digitalWrite(13, LOW);        //Turning pin 13 off
          }
